# 预测交互数据API

```json
// 数据格式
{
    msg: 1, // 1.success 2. bad
    data: {
        time: ,
        list: [
            {
        		id:  竞猜id
        		base_id: 根据比赛的基础id
        		type: 'game'， 竞猜的类型：目前只有比赛
        		home_team: 主队
        		grust_team: 客队
        		system_res: 系统认为
        		approve_sys: 已赞成系统的声望值
        		oppose_sys: 已反对系统的声望值
        		last_tiem: 截止日期
            }
        ]
    }
}

// 提交数据格式
{
    id: 竞猜id,
    aid: 用户id,
    select： 选择，
    value: 押注的声望，
}

// 提交数据可能返回的数据
{
    status： 1ok, 0 No
    msg: '成功/已押注/声望不足' 
}

```
